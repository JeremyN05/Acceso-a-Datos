			ObjectInputStream ois = new ObjectInputStream(new FileInputStream("empleado.dat"));
			
			ArrayList<Empleados> empleados = (ArrayList<Empleados>) ois.readObject();
			
			System.out.println("Las personas almacenadas en el fichero son: ");
			
			for(Empleados p : empleados) {
				
				System.out.println(p);
				
			}
			
		}catch(IOException | ClassNotFoundException e) {
			
			e.printStackTrace();
			
		}
		
	}
